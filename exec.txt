0.7  Build libhypertrace.so — The Native JVMTI Agent [NEW — Replaces Frida]
NEW PREREQUISITE — Build the Native JVMTI Agent
This step replaces the entire Frida dependency. libhypertrace.so is a custom JVMTI
agent compiled from scratch using the Android NDK. It is baked directly into the
QEMU AVD system image, making it indistinguishable from a system library to any

Step 1 — Write the JVMTI Agent Source (jvmti_agent.cpp):
// File: ~/analysis/jvmti_agent/jvmti_agent.cpp
// Custom JVMTI agent for method entry/exit + JNI boundary tracing
// Compiles to libhypertrace.so — no Frida, no external sockets, no gmain thread
 
#include <jvmti.h>
#include <dlfcn.h>       // for dlopen/dlsym JNI boundary hooks
#include <fstream>
#include <chrono>
#include <sstream>
#include <mutex>
 
static std::ofstream log_file;
static std::mutex    log_mutex;
 
// Nanosecond timestamp using CLOCK_MONOTONIC_RAW (same source as strace)
static jlong now_ns() {
  struct timespec ts;
  clock_gettime(CLOCK_MONOTONIC_RAW, &ts);
  return (jlong)ts.tv_sec * 1000000000LL + ts.tv_nsec;
}
 
void JNICALL MethodEntry(jvmtiEnv* jvmti, JNIEnv* jni,
                         jthread thread, jmethodID method) {
  char *name, *sig, *generic;
  jvmti->GetMethodName(method, &name, &sig, &generic);
  jclass klass; jvmti->GetMethodDeclaringClass(method, &klass);
  char *class_sig; jvmti->GetClassSignature(klass, &class_sig, nullptr);
  jlong tid_storage = 0;
  jvmti->GetThreadLocalStorage(thread, (void**)&tid_storage);
  std::lock_guard<std::mutex> lk(log_mutex);
  log_file << "ENTRY|" << now_ns() << "|" << tid_storage
           << "|" << class_sig << "|" << name << "|" << sig << "\n";
  jvmti->Deallocate((unsigned char*)name);
  jvmti->Deallocate((unsigned char*)sig);
  jvmti->Deallocate((unsigned char*)class_sig);
}
 
void JNICALL MethodExit(jvmtiEnv* jvmti, JNIEnv* jni, jthread thread,
                        jmethodID method, jboolean had_exception,
                        jvalue return_value) {
  char *name, *sig, *generic;
  jvmti->GetMethodName(method, &name, &sig, &generic);
  jclass klass; jvmti->GetMethodDeclaringClass(method, &klass);
  char *class_sig; jvmti->GetClassSignature(klass, &class_sig, nullptr);
  jlong tid_storage = 0;
  jvmti->GetThreadLocalStorage(thread, (void**)&tid_storage);
  std::lock_guard<std::mutex> lk(log_mutex);
  log_file << "EXIT|" << now_ns() << "|" << tid_storage
           << "|" << class_sig << "|" << name << "|" << sig << "\n";
  jvmti->Deallocate((unsigned char*)name);
  jvmti->Deallocate((unsigned char*)sig);
  jvmti->Deallocate((unsigned char*)class_sig);
}
 
// JNI boundary hook: fires when ART loads a new class
// Walks the loaded .so export table using dlsym — no external process needed
void JNICALL ClassLoad(jvmtiEnv* jvmti, JNIEnv* jni,
                       jthread thread, jclass klass) {
  char *class_sig; jvmti->GetClassSignature(klass, &class_sig, nullptr);
  std::lock_guard<std::mutex> lk(log_mutex);
  log_file << "CLASSLOAD|" << now_ns() << "|" << class_sig << "\n";
  jvmti->Deallocate((unsigned char*)class_sig);
}
 
JNIEXPORT jint JNICALL Agent_OnLoad(JavaVM* vm, char* options, void* reserved) {
  log_file.open("/sdcard/jvmti_trace.log", std::ios::app);
  jvmtiEnv* jvmti;
  vm->GetEnv((void**)&jvmti, JVMTI_VERSION_1_2);
  jvmtiCapabilities caps = {};
  caps.can_generate_method_entry_events = 1;
  caps.can_generate_method_exit_events  = 1;
  caps.can_generate_class_load_events   = 1;   // for JNI boundary tracing
  jvmti->AddCapabilities(&caps);
  jvmtiEventCallbacks cb = {};
  cb.MethodEntry = MethodEntry;
  cb.MethodExit  = MethodExit;
  cb.ClassLoad   = ClassLoad;
  jvmti->SetEventCallbacks(&cb, sizeof(cb));
  jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_METHOD_ENTRY, nullptr);
  jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_METHOD_EXIT,  nullptr);
  jvmti->SetEventNotificationMode(JVMTI_ENABLE, JVMTI_EVENT_CLASS_LOAD,   nullptr);
  return JNI_OK;
}

Step 3 — Install APK and Attach Native JVMTI Agent via ART Plugin
This is the core change replacing Frida injection. The native JVMTI agent is loaded via Android's built-in 'wrap' system property, which instructs the Android Runtime to load additional JVMTI plugins before the application's first bytecode instruction executes. This is architecturally identical to how Android Studio's profiler attaches — no external socket, no secondary process, no detectable artifacts.

# Get the app's package name from AndroidManifest

# e.g.: package: name='com.malicious.app'

# This uses Android's native wrap mechanism — the same used by Android Studio profiler
# It loads the agent into ART's address space before any app bytecode runs
adb shell setprop wrap.com.malicious.app \
    'logwrapper /system/bin/app_process -Xplugin:/system/lib64/libhypertrace.so'
 
# Launch the application — agent is already in ART when first instruction runs
adb shell am start -n com.malicious.app/.MainActivity
 
# Give app 3 seconds to initialise before starting strace
sleep 3
 
# Verify agent is producing output
adb shell wc -l /sdcard/jvmti_trace.log
# Expected: line count > 0 within 5 seconds of launch

Why This Is Undetectable (vs Frida)
Frida creates: a server socket on port 27042, a gmain thread from GLib, and
writes frida-agent.so to /proc/self/maps. All three are well-known detection vectors.

libhypertrace.so creates: nothing detectable. It is already in /system/lib64/
(looks like a system library), loads via ART's own plugin mechanism, uses only
the JVMTI API that ART already exposes to its own profiling subsystem, and has
no sockets, no external threads, and no identifiable network activity.

Step 2 — Write and Run the Fuzzer (with DFS Depth Cap + Coverage Logging)
# File: semantic_fuzzer.py
import uiautomator2 as u2
import time, re, subprocess, json
 
d = u2.connect()  # connects to running emulator
 
TEXT_PATTERNS = {
    r'(email|mail)':    'test.user@gmail.com',
    r'(password|pass)': 'TestPass123!',
    r'(phone|mobile)':  '9876543210',
    r'(otp|pin|code)':  '123456',
    r'(name|user)':     'John Doe',
}
BUTTON_PATTERNS = re.compile(
    r'(agree|continue|next|ok|allow|accept|skip|done|confirm|yes)',
    re.IGNORECASE)
 
visited_activities = []
MAX_DEPTH    = 6   # FIX: DFS depth cap — prevents infinite traversal on dynamic layouts
POLL_HZ      = 10
 
def get_current_activity():
    result = subprocess.run(['adb','shell','dumpsys','activity','activities'],
        capture_output=True, text=True)
    for line in result.stdout.splitlines():
        if 'mCurrentFocus' in line:
            return line.strip()
    return ''
 
def fuzz_once(depth=0):
    if depth > MAX_DEPTH:
        return  # DFS depth cap enforced here
    for elem in d.xpath('//android.widget.EditText').all():
        desc = (elem.attrib.get('content-desc','') +
                elem.attrib.get('resource-id','') +
                elem.attrib.get('hint','')).lower()
        for pattern, value in TEXT_PATTERNS.items():
            if re.search(pattern, desc):
                elem.set_text(value)
                break
    for elem in d.xpath('//android.widget.Button').all():
        text = elem.attrib.get('text','')
        if BUTTON_PATTERNS.search(text):
            elem.click()
            print(f'  [fuzzer] clicked: {text}')
            time.sleep(0.3)
 
def run_fuzzer(duration=35):
    end_time = time.time() + duration
    while time.time() < end_time:
        current = get_current_activity()
        # FIX: Cycle detection — if stuck at same activity for 2+ cycles, press back
        if visited_activities[-2:].count(current) >= 2:
            d.press('back')
            time.sleep(0.5)
        else:
            visited_activities.append(current)
            fuzz_once()
        time.sleep(1.0 / POLL_HZ)
    # FIX: Save coverage log for paper evaluation
    unique_activities = list(set(visited_activities))
    json.dump({'visited': visited_activities, 'unique': unique_activities},
              open('~/analysis/logs/fuzzer_coverage.json','w'))
    print(f'[fuzzer] Done. Visited {len(unique_activities)} unique Activities.')
 
if __name__ == '__main__':
    run_fuzzer(duration=35)

import sqlite3, re
 
def parse_jvmti(path):
    events = []
    for line in open(path):
        parts = line.strip().split('|')
        # Format: ENTRY|<ts_ns>|<tid>|<class_sig>|<method>|<sig>
        # Format: EXIT|<ts_ns>|<tid>|<class_sig>|<method>|<sig>
        # Format: CLASSLOAD|<ts_ns>|<class_sig>  (NEW — from ClassLoad callback)
        if len(parts) == 6 and parts[0] in ('ENTRY','EXIT'):
            events.append({
                'type':       parts[0],
                'ts_ns':      int(parts[1]),
                'tid':        int(parts[2]),
                'class_sig':  parts[3],
                'method':     parts[4],
                'sig':        parts[5],
            })
        elif len(parts) == 3 and parts[0] == 'CLASSLOAD':
            events.append({
                'type':       'CLASSLOAD',
                'ts_ns':      int(parts[1]),
                'class_sig':  parts[2],
                'tid':        0, 'method':'', 'sig':'',
            })
    return events

# File: parse_strace.py
import re
 
SYSCALL_RE = re.compile(
    r'(\d+)\s+([\d.]+)\s+(\w+)\((.*)\)\s+=\s+(.+)'
)
 
def parse_strace(path):
    syscalls = []
    for line in open(path, errors='ignore'):
        m = SYSCALL_RE.match(line.strip())
        if m:
            syscalls.append({
                'tid':       int(m.group(1)),
                'ts_ns':     int(float(m.group(2)) * 1e9),
                'syscall':   m.group(3),
                'args':      m.group(4),
                'retval':    m.group(5).strip(),
            })
    return syscalls
 
strace_events = parse_strace('~/analysis/logs/strace.log')

# File: ssaf_filter.py
import json, re
 
# Security-sensitive API pattern taxonomy
# Methods matching ANY of these patterns are routed to the LLM-Describer
SENSITIVE_PATTERNS = [
    # Dynamic code loading
    r'DexClassLoader', r'PathClassLoader', r'loadClass', r'defineClass',
    # Cryptographic operations
    r'javax\.crypto', r'AES', r'RSA', r'KeyGenerator', r'SecretKeySpec',
    # SMS / Telephony
    r'SmsManager', r'sendTextMessage', r'TelephonyManager', r'subscriberId',
    # Process / shell execution
    r'Runtime\.exec', r'ProcessBuilder', r'execve',
    # Reflection / dynamic dispatch
    r'java\.lang\.reflect', r'getDeclaredMethod', r'invoke\(',
    # Network with explicit targets
    r'HttpURLConnection', r'OkHttpClient', r'Socket\(', r'DatagramSocket',
    # Persistence mechanisms
    r'BOOT_COMPLETED', r'AlarmManager', r'JobScheduler',
    # Privilege escalation
    r'\bsu\b', r'SELinux', r'AccessibilityService', r'DeviceAdminReceiver',
    # Native library loading
    r'System\.loadLibrary', r'System\.load\(',
    # Security-sensitive syscalls in kernel log
    r'openat', r'connect', r'execve', r'mmap',
]
 
SENSITIVE_RE = re.compile('|'.join(SENSITIVE_PATTERNS))
 
sensitive_chunks = []
benign_chunks    = []
 
for line in open('~/analysis/chunks.jsonl'):
    chunk = json.loads(line)
    jimple   = chunk.get('raw_jimple', '')
    syscalls = ' '.join(s.get('syscall','') + ' ' + s.get('args','')
                        for s in chunk.get('kernel_syscalls',[]))
    combined = jimple + ' ' + syscalls
 
    if SENSITIVE_RE.search(combined):
        chunk['ssaf_flagged'] = True
        sensitive_chunks.append(chunk)
    else:
        chunk['ssaf_flagged'] = False
        # Benign chunks get a raw stub as their summary — no LLM call
        chunk['summary'] = chunk.get('raw_jimple','')[:200]
        benign_chunks.append(chunk)
 
with open('~/analysis/chunks_sensitive.jsonl', 'w') as f:
    for c in sensitive_chunks:
        f.write(json.dumps(c) + '\n')
 
with open('~/analysis/chunks_benign.jsonl', 'w') as f:
    for c in benign_chunks:
        f.write(json.dumps(c) + '\n')
 
total = len(sensitive_chunks) + len(benign_chunks)
pct = 100 * len(sensitive_chunks) / max(1, total)
print(f'SSAF complete:')
print(f'  Sensitive (-> LLM):  {len(sensitive_chunks)} chunks ({pct:.1f}%)')
print(f'  Benign (raw stub):   {len(benign_chunks)} chunks ({100-pct:.1f}%)')

SYSTEM_PROMPT = '''You are a malware analyst. You will receive a batch of
Java method execution records from a sandboxed Android application.
For each METHOD CHUNK, write one concise paragraph (2-4 sentences) describing
what the method ACTUALLY DOES at the OS/behavioural level.
Ignore obfuscated names. Focus on syscalls, API calls, and data flows.

Output one JSON object per line. No markdown, no preamble.'''
 
def build_batch_prompt(chunks):
    parts = []
    for c in chunks:
        syscall_summary = [f"{s['syscall']}({s['args'][:60]})"

            f'JIMPLE (truncated):\n{c["raw_jimple"][:800]}\n'
            f'KERNEL SYSCALLS: {"; ".join(syscall_summary)}\n'

Layer 3.2 — DERP + Call-Graph Prefetch & CFG-Guided Pruning

CRITICAL REVISION — Dynamic-Edge Recovery Pass (DERP) Added Before Pruning
The original prefetch_and_prune.py ran prune_jimple() directly on static CFG labels.
For packed/obfuscated/reflective apps, Soot's static CFG has disconnected nodes.
Pruning against an incomplete CFG silently discards the malicious execution paths.

The fix: dynamic_edge_recovery.py runs FIRST. It compares executed_cfg (from JVMTI)
against static Soot labels, synthesises Execution Edges for runtime-only paths,
and produces an augmented label set. Pruning then runs against the augmented set.

Step 1 — Write the DERP Engine (dynamic_edge_recovery.py) [NEW]
# File: dynamic_edge_recovery.py  — Dynamic-Edge Recovery Pass
# Must run BEFORE prune_jimple() in prefetch_and_prune.py
import sqlite3, json
 
db = sqlite3.connect('~/analysis/pipeline.db')
 
def get_static_labels_for_method(method_name, class_name):
    """Retrieve the Soot-known CFG branch labels for this method."""
    row = db.execute(
        'SELECT cfg_json FROM jimple_methods WHERE class_name=? AND method_name=?',
        (class_name, method_name)).fetchone()
    if not row or not row[0]:
        return set()
    cfg = json.loads(row[0])
    return set(cfg.get('labels', []))
 
def recover_dynamic_edges(chunk):
    """
    Compare JVMTI-observed branches against static Soot CFG labels.
    For any branch observed at runtime but absent from static CFG,
    synthesise an Execution Edge and record it in the dynamic_edges table.
    Returns: (augmented_labels, recovered_edges)
    """
    method     = chunk.get('method_name', '')
    class_name = chunk.get('class_name', '')
    executed   = set(chunk.get('executed_cfg', []))
    static     = get_static_labels_for_method(method, class_name)
 
    # Branches seen at runtime but invisible to Soot
    dynamic_only = executed - static
 
    recovered_edges = []
    for label in dynamic_only:
        try:
            db.execute(
                'INSERT OR IGNORE INTO dynamic_edges (method, class, label, type, ts) VALUES (?,?,?,?,?)',
                (method, class_name, label, 'DYNAMIC_ONLY', chunk.get('entry_ts_ns', 0))
            )
            recovered_edges.append({
                'method': method, 'class': class_name,
                'label': label, 'type': 'DYNAMIC_ONLY',
            })
        except Exception as e:
            pass
 
    db.commit()
 
    # Augmented set: static labels UNION dynamic-only labels
    augmented = static.union(dynamic_only)
    return augmented, recovered_edges

Step 3 — Prefetch Call Chain (Depth=2)
def prefetch_call_chain(seed_method, depth=2):
    neighbors = set([seed_method])
    for _ in range(depth):
        new_nodes = set()
        for n in list(neighbors):
            new_nodes.update(G.predecessors(n))
            new_nodes.update(G.successors(n))
        neighbors.update(new_nodes)
    return list(neighbors)
 
from qdrant_client import QdrantClient
from qdrant_client.models import Filter, FieldCondition, MatchAny
 
qdrant = QdrantClient(host='localhost', port=6333)
 
def batch_fetch_methods(method_names):
    results = qdrant.scroll(
        collection_name='malware_traces',
        scroll_filter=Filter(must=[FieldCondition(
            key='method_name', match=MatchAny(any=method_names))]),
        limit=len(method_names)+10, with_payload=True
    )
    return [r.payload for r in results[0]]

Step 4 — CFG Pruning (Revised: runs over DERP-augmented labels)
def prune_jimple(jimple_text, augmented_labels):
    """
    Remove lines belonging to unexecuted CFG branches.
    augmented_labels = static Soot labels UNION DERP-recovered dynamic labels.
    This ensures dynamically-discovered paths are never pruned.
    """
    if not augmented_labels:
        return jimple_text[:1500]  # fallback truncation for no-CFG methods
    lines  = jimple_text.splitlines()
    kept   = []
    keep   = True
    for line in lines:
        stripped = line.strip()
        if stripped.endswith(':') and stripped[:-1] not in augmented_labels:
            keep = False   # entering an unexecuted block (not in augmented set)
        if stripped.endswith(':') and stripped[:-1] in augmented_labels:
            keep = True    # re-entering an executed or DERP-recovered block
        if keep:
            kept.append(line)
    return '\n'.join(kept)

Step 2 — Three-Condition Verifier (Replaces Existence-Only Check)
CRITICAL FIX — Three-Condition Causal Verification
Condition 1: Syscall existence — claimed syscall must appear in strace log.
Condition 2: Thread ID match — syscall's kernel TID must equal JVMTI Java TID.
             An LLM cannot fabricate this — TIDs are raw integers from two independent sources.
Condition 3: Temporal containment — syscall timestamp must fall within [MethodEntry_ts, MethodExit_ts].
             This proves causality, not just coincidence.
All 3 conditions must pass for a claim to be VERIFIED.

# File: verify_claims.py
import json, sqlite3
from parse_strace import parse_strace
 
strace_events = parse_strace('~/analysis/logs/strace.log')
db = sqlite3.connect('~/analysis/pipeline.db')
 
def get_method_window(method_name, class_name):
    row = db.execute(
        'SELECT entry_ts_ns, exit_ts_ns, thread_id FROM method_windows WHERE method=? AND class=?',
        (method_name, class_name)).fetchone()
    return row  # (entry_ts, exit_ts, tid) or None
 
def verify_claim(claim):
    method   = claim.get('method_name', '')
    syscall  = claim.get('claimed_syscall', '')
    resource = claim.get('claimed_resource', '')
 
    window = get_method_window(method, '')
    if not window:
        return 'UNVERIFIED', 'No method window found in JVMTI logs'
 
    entry_ts, exit_ts, jvmti_tid = window
 
    # Condition 1: Syscall existence
    matching = [s for s in strace_events
                if syscall.lower() in s['syscall'].lower()
                and resource in s.get('args','')]
    if not matching:
        return 'UNVERIFIED', f'Syscall {syscall}({resource}) not in strace'
 
    # Condition 2: Thread ID match
    tid_matched = [s for s in matching if s['tid'] == jvmti_tid]
    if not tid_matched:
        return 'PARTIAL', f'Syscall exists but TID mismatch (JVMTI TID={jvmti_tid})'
 
    # Condition 3: Temporal containment
    temporally_matched = [s for s in tid_matched
                          if entry_ts <= s['ts_ns'] <= exit_ts]
    if not temporally_matched:
        return 'PARTIAL', 'Syscall TID matches but timestamp outside method window'
 
    return 'VERIFIED', f'All 3 conditions passed — {len(temporally_matched)} matching events'

import json
 
SECURITY_SYSCALLS = {'openat','connect','sendto','execve','mmap','clone'}
 
def count_sensitive_syscalls(strace_log_path):
    count = 0
    for line in open(strace_log_path, errors='ignore'):
        if any(s in line for s in SECURITY_SYSCALLS):
            count += 1
    return count
 
def behavioral_extraction_gain(control_log, hardened_log):
    E_control  = count_sensitive_syscalls(control_log)
    E_hardened = count_sensitive_syscalls(hardened_log)
    E_total    = max(E_hardened, 1)
    delta_B    = (E_hardened - E_control) / E_total
    return delta_B, E_control, E_hardened
 
results = []

        sample['control_strace'], sample['hardened_strace'])

                    'E_control': E_ctrl, 'E_hardened': E_hard})
    print(f'{sample["name"]}: ΔB={delta_B:.3f} (control={E_ctrl}, hardened={E_hard})')
